from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait

from bs4 import BeautifulSoup
from time import gmtime, strftime
import time

url = "https://paginawebparascraping.com/"

# Configuraci√≥n del driver
driver = webdriver.Chrome()
driver.get(url)

# Esperar que la p√°gina cargue completamente
time.sleep(10)

# Realizar scroll hacia abajo
scroll_pause_time = 5  # Tiempo de espera entre scrolls

# Hacer desplazamientos hacia abajo
for _ in range(20):
    # Hacer scroll hacia abajo
    driver.execute_script("window.scrollTo(0, document.body.scrollHeight);")
    
    # Esperar para que se carguen nuevos elementos
    time.sleep(scroll_pause_time)

# Extracci√≥n de info
soup = BeautifulSoup(driver.page_source, "html.parser")
articles = soup.find_all("article")
print(f"Found {len(articles)} articles")  # Confirmar cantidad de art√≠culos

scraped_data = []
for article in articles:
    # Extraer los datos
    link_tag = article.find("a", class_="bodytext content")
    link = link_tag["href"] if link_tag else "No link found"

    title_tag = article.find("h2")
    title = title_tag.get_text(strip=True) if title_tag else "No title found"

    date_tag = article.find("time")
    date = date_tag["datetime"] if date_tag and date_tag.has_attr("datetime") else "No date found"

    # Imprimir datos para depuraci√≥n
    print(f"Link: {link}")
    print(f"Title: {title}")
    print(f"Date: {date}")

    # Agregar a la lista solo si todos los datos est√°n disponibles
    if link != "No link found" and title != "No title found" and date != "No date found":
        scraped_data.append({"title": title, "link": link, "date": date})

# Guardar los datos en un archivo .txt
fecha = strftime("%Y-%m-%d %H-%M-%S", gmtime())
file_name = f"file_scraped_data_{fecha}.txt"

with open(file_name, "w", encoding="utf-8") as file:
    for entry in scraped_data:
        file.write(f"Title: {entry['title']}\n")
        file.write(f"Date: {entry['date']}\n")
        file.write(f"Link: {entry['link']}\n")
        file.write("\n")  # L√≠nea vac√≠a entre noticias

# Mensaje final
print(f"Datos guardados correctamente en '{file_name}'")

# Cerrar el navegador
driver.quit()
